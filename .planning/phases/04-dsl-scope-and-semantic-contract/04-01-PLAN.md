<!-- Existing runtime concepts the authored DSL must map onto -->

pub trait Rule<S, O, E, P>
where
    S: Clone,
    O: Operation<S>,
    P: Copy + Ord,
{
    fn id(&self) -> &'static str;
    fn priority(&self) -> P;
    fn before(&self, _state: &S, _event: &mut E, _tx: &mut Transaction<O>) {}
    fn after(&self, _state: &S, _event: &E, _tx: &mut Transaction<O>) {}
}

pub enum RuleLifetime {
    Permanent,
    Turns(u32),
    Triggers(u32),
}

pub struct Transaction<O> {
    pub ops: Vec<O>,
    pub irreversible: bool,
    pub deterministic: bool,
    pub cancelled: bool,
}
