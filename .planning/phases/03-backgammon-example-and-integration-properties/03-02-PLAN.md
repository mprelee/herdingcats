```rust
// Engine construction
Engine::<S, O, E, P>::new(state: S) -> Engine<S, O, E, P>
engine.add_rule(rule: impl Rule<S,O,E,P>, lifetime: RuleLifetime)
engine.dispatch(event: E, tx: Transaction<O>)
engine.dispatch_preview(event: E, tx: Transaction<O>)
engine.undo() -> bool
engine.read() -> S
engine.replay_hash() -> u64
engine.state: S  // direct field access

// Rule trait
pub trait Rule<S, O, E, P>: ... {
    fn id(&self) -> &'static str;
    fn priority(&self) -> P;
    fn before(&self, state: &S, event: &mut E, tx: &mut Transaction<O>) { }  // default no-op
    fn after(&self, state: &S, event: &E, tx: &mut Transaction<O>) { }       // default no-op
}

// Transaction
Transaction::<O>::new()  // irreversible: true (default), cancelled: false
tx.irreversible: bool    // set to false for non-undoable events
tx.cancelled: bool
tx.ops: Vec<O>           // push ops here in rule.before()/after()

// RuleLifetime variants used here
RuleLifetime::Permanent
```

Critical semantics (BACK-03, from research):
```
tx.irreversible = false  → engine applies ops but does NOT push CommitFrame → engine.undo() skips it
tx.irreversible = true   → engine pushes CommitFrame → engine.undo() can reverse it
```

    ```rust
    #[cfg(test)]
    mod props {
        use super::*;
        use proptest::prelude::*;

        // Strategy: generate a single valid MoveOp from standard starting position.
        // Generate from a set of known-valid source points (indices with White checkers).
        // Valid White source points in starting position: 5, 7, 12, 23
        // Move distance constrained 1..=6 (standard die range), to must be < 24 and >= 0.
        prop_compose! {
            fn valid_move_op_strategy()
                (from_choice in prop_oneof![Just(5usize), Just(7usize), Just(12usize), Just(23usize)],
                 distance in 1usize..=6,
                 die_index in 0usize..=1usize)
            -> BackgammonOp {
                // to = from - distance for White (moves toward lower indices in 0-indexed)
                // Clamp to avoid underflow; use prop_assume! in the test if needed
                let to = from_choice.saturating_sub(distance);
                BackgammonOp::MoveOp {
                    from: from_choice,
                    to,
                    captured: false,  // standard position has no opposing blots at generated positions
                    die_index,
                }
            }
        }

        // Strategy: generate a sequence of conservative ops for BACK-05.
        // Use only MoveOp with from and to both on the board (0..=23), captured=false.
        // Apply and immediately undo each op to maintain conservation without
        // needing valid game state continuity across ops.
        prop_compose! {
            fn conservative_op_sequence()
                (ops in proptest::collection::vec(
                    (0usize..24, 1usize..=6, 0usize..=1usize).prop_map(|(from, dist, die)| {
                        BackgammonOp::MoveOp {
                            from,
                            to: from.saturating_sub(dist),
                            captured: false,
                            die_index: die,
                        }
                    }),
                    0..=20
                ))
            -> Vec<BackgammonOp> { ops }
        }

        proptest! {
            // BACK-05: board conservation invariant
            #[test]
            fn prop_board_conservation(ops in conservative_op_sequence()) {
                let mut state = BgState::new();
                let count_before = checker_count(&state);
                prop_assert_eq!(count_before, 30u32);

                for op in &ops {
                    op.apply(&mut state);
                    op.undo(&mut state);
                }
                prop_assert_eq!(checker_count(&state), 30u32);
            }

            // BACK-06: per-die undo roundtrip
            #[test]
            fn prop_per_die_undo(op in valid_move_op_strategy()) {
                let mut engine = Engine::<BgState, BackgammonOp, BackgammonEvent, BackgammonPriority>::new(BgState::new());
                engine.add_rule(RollDiceRule, RuleLifetime::Permanent);
                engine.add_rule(MoveRule, RuleLifetime::Permanent);

                // First roll dice (non-undoable — sets up dice state)
                {
                    let mut tx = Transaction::new();
                    tx.irreversible = false;
                    engine.dispatch(BackgammonEvent::RollDice { d1: 3, d2: 5 }, tx);
                }

                let state_before = engine.read();
                let hash_before = engine.replay_hash();

                // Dispatch a Move using the generated op's from/to/die_index
                if let BackgammonOp::MoveOp { from, to, die_index, .. } = op {
                    engine.dispatch(
                        BackgammonEvent::Move { from, to, die_index },
                        Transaction::new(),
                    );
                    engine.undo();
                    prop_assert_eq!(engine.read(), state_before);
                    prop_assert_eq!(engine.replay_hash(), hash_before);
                }
            }
        }
    }
    ```

    IMPORTANT NOTES for prop_per_die_undo:
    - The proptest strategy uses valid White starting positions; some generated moves may result in board[to] underflow if the from point is empty after the dice roll state setup. Use prop_assume!(state.board[from] > 0) after reading state_before, OR ensure the strategy only picks from known-populated points. Prefer the prop_assume! approach so the strategy stays simple.
    - Both assertions (engine.read() and engine.replay_hash()) are REQUIRED — state-only assertion is insufficient (established Phase 2 pattern, locked in from RESEARCH.md).

    IMPORTANT NOTES for prop_board_conservation:
    - Uses apply+undo pairs to ensure conservation holds through roundtrips — this tests that ops are conservative even when not in a valid game sequence.
    - This is consistent with CONTEXT.md decision: "op-level generation: bypasses event/rule layer, focused on conservation invariant".
